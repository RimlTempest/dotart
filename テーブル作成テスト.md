USE dotart_db;
/* テーブル作成 */
CREATE TABLE IF NOT EXISTS users (
  id INT NOT NULL AUTO_INCREMENT,
  user_id VARCHAR(256) NOT NULL UNIQUE,
  display_name VARCHAR(256) NOT NULL,
  email VARCHAR(256) NOT NULL UNIQUE,
  pass_word VARCHAR(256) NOT NULL,
  last_name VARCHAR(256) NOT NULL,
  first_name VARCHAR(256) NOT NULL,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY(id, user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;CREATE TABLE IF NOT EXISTS canvases (
  id INT NOT NULL AUTO_INCREMENT,
  canvas_id VARCHAR(256) NOT NULL UNIQUE,
  user_id VARCHAR(256) NOT NULL,
  canvas_name VARCHAR(256) NOT NULL,
  canvas_range VARCHAR(256),
  pallet TEXT,
  index_data TEXT,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY(id, canvas_id),
  FOREIGN KEY (user_id)
    REFERENCES users(user_id)
    ON DELETE CASCADE
    ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8;CREATE TABLE IF NOT EXISTS basic_pallet (
  id INT NOT NULL AUTO_INCREMENT,
  pallet_id VARCHAR(256) NOT NULL UNIQUE,
  pallet_name VARCHAR(256) NOT NULL,
  summary VARCHAR(256),
  pallet_data TEXT,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY(id, pallet_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;CREATE TABLE IF NOT EXISTS users_pallet (
  id INT NOT NULL AUTO_INCREMENT,
  pallet_id VARCHAR(256) NOT NULL UNIQUE,
  user_id VARCHAR(256) NOT NULL,
  pallet_name VARCHAR(256),
  pallet_data TEXT,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  updated_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  PRIMARY KEY(id, pallet_id),
  FOREIGN KEY (user_id)
    REFERENCES users(user_id)
    ON DELETE CASCADE
    ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
